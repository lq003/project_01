//检查文件状态
    #详细列表
    git status 
    #精简状态
    git status -s

注释:标准的工作流程 工作区→暂存区→Git仓库
//文件操作
    #暂存文件
    git add 文件名.后缀
    #暂存所有文件
    git add .

    #取消暂存的文件
    git reset HEAD 文件名.后缀
    #撤销所有暂存文件
    git reset HEAD .
    
    #提交更新，-m选项后面是用来对提交的内容做进一步的描述
    git commit -m "描述信息"

    #撤销文件修改，还原成Git仓库中所保存的版本
    git checkout -- 文件名.后缀

    #跳过暂存区，直接提交Git仓库，工作区→Git仓库
    git commit -a -m "描述信息"
